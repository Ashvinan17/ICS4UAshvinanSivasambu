fastevent.py 
1    """ 
2    A compatibility shim for pygame.fastevent based on pygame.event. 
3    This module was deprecated in pygame 2.2, and is scheduled for removal in a 
4    future pygame version. If you are using pygame.fastevent, please migrate to 
5    using regular pygame.event module 
6    """
7    
8    import pygame.event
9    import pygame.display
10   from pygame import error, register_quit
11   from pygame.event import Event
12   
13   _ft_init = False
14   
15   
16   def _ft_init_check():
17       """ 
18       Raises error if module is not init 
19       """
20       if not _ft_init:
21           raise error("fastevent system not initialized")
22   
23   
24   def _quit_hook():
25       """ 
26       Hook that gets run to quit module 
27       """
28       global _ft_init
29       _ft_init = False
30   
31   
32   def init():
33       """init() -> None 
34       initialize pygame.fastevent 
35       """
36       global _ft_init
37       if not pygame.display.get_init():
38           raise error("video system not initialized")
39   
40       register_quit(_quit_hook)
41       _ft_init = True
42   
43   
44   def get_init():
45       """get_init() -> bool 
46       returns True if the fastevent module is currently initialized 
47       """
48       return _ft_init
49   
50   
51   def pump():
52       """pump() -> None 
53       internally process pygame event handlers 
54       """
55       _ft_init_check()
56       pygame.event.pump()
57   
58   
59   def wait():
60       """wait() -> Event 
61       wait for an event 
62       """
63       _ft_init_check()
64       return pygame.event.wait()
65   
66   
67   def poll():
68       """poll() -> Event 
69       get an available event 
70       """
71       _ft_init_check()
72       return pygame.event.poll()
73   
74   
75   def get():
76       """get() -> list of Events 
77       get all events from the queue 
78       """
79       _ft_init_check()
80       return pygame.event.get()
81   
82   
83   def post(event: Event):
84       """post(Event) -> None 
85       place an event on the queue 
86       """
87       _ft_init_check()
88       pygame.event.post(event)
