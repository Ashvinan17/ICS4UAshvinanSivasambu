17   """Test the Database class."""
18   
19   from unittest import TestCase
20   from worldlib.database import Database
21   
22   
23   class TestDatabase(TestCase):
24       def test_basic(self):
25           db = Database()
26           self.assertEqual(db.lookup_code('it'), 'Italy')
27   
28       def test_find_matches(self):
29           db = Database()
30           matches = db.find_matches('italy')
31           self.assertEqual(len(matches), 1)
32           self.assertEqual(matches[0], ('it', 'Italy'))
33   
34       def test_find_matches_uppercase(self):
35           db = Database()
36           matches = db.find_matches('ITALY')
37           self.assertEqual(len(matches), 1)
38           self.assertEqual(matches[0], ('it', 'Italy'))
39   
40       def test_find_matches_multiple(self):
41           db = Database()
42           matches = db.find_matches('united')
43           self.assertEqual(len(matches), 6)
44           self.assertEqual(sorted(matches), [
45               ('ae', 'United Arab Emirates (the)'),
46               ('gb',
47                'United Kingdom of Great Britain and Northern Ireland (the)'),
48               ('tz', 'Tanzania, United Republic of'),
49               ('uk', 'United Kingdom (common practice)'),
50               ('um', 'United States Minor Outlying Islands (the)'),
51               ('us', 'United States of America (the)'),
52               ])
53   
54       def test_iteration(self):
55           codes = []
56           for code in Database():
57               codes.append(code)
58           top_5 = sorted(codes)[:5]
59           self.assertEqual(top_5, ['ac', 'ad', 'ae', 'aero', 'af'])
