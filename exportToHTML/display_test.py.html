362      def test_iconify(self):
363          pygame.display.set_mode((640, 480))
364  
365          self.assertEqual(pygame.display.get_active(), True)
366  
367          success = pygame.display.iconify()
368  
369          if success:
370              active_event = window_minimized_event = False
371              # make sure we cycle the event loop enough to get the display
372              # hidden. Test that both ACTIVEEVENT and WINDOWMINIMISED event appears
373              for _ in range(50):
374                  time.sleep(0.01)
375                  for event in pygame.event.get():
376                      if event.type == pygame.ACTIVEEVENT:
377                          if not event.gain and event.state == pygame.APPACTIVE:
378                              active_event = True
379                      if event.type == pygame.WINDOWMINIMIZED:
380                          window_minimized_event = True
381  
382              self.assertTrue(window_minimized_event)
383              self.assertTrue(active_event)
384              self.assertFalse(pygame.display.get_active())
385  
386          else:
387              self.fail("Iconify not supported on this platform, please skip")

488      def test_set_gamma_ramp(self):
489  
490          # __doc__ (as of 2008-08-02) for pygame.display.set_gamma_ramp:
491  
492          # change the hardware gamma ramps with a custom lookup
493          # pygame.display.set_gamma_ramp(red, green, blue): return bool
494          # set_gamma_ramp(red, green, blue): return bool
495          #
496          # Set the red, green, and blue gamma ramps with an explicit lookup
497          # table. Each argument should be sequence of 256 integers. The
498          # integers should range between 0 and 0xffff. Not all systems and
499          # hardware support gamma ramps, if the function succeeds it will
500          # return True.
501          #
502          pygame.display.set_mode((5, 5))
503          r = list(range(256))
504          g = [number + 256 for number in r]
505          b = [number + 256 for number in g]
506          isSupported = pygame.display.set_gamma_ramp(r, g, b)
507          if isSupported:
508              self.assertTrue(pygame.display.set_gamma_ramp(r, g, b))
509          else:
510              self.assertFalse(pygame.display.set_gamma_ramp(r, g, b))
